```python
def _with_rank_limit(rows: list[dict[str, Any]], limit: int = 10) -> list[dict[str, Any]]:
    ranked_rows = []
    for index, row in enumerate(rows[:limit], start=1):
        ranked_rows.append({
            **row,
            'rank': index,
        })
    return ranked_rows
```
